示例代码 #

vue
<script setup>
  const tableSet = {
    api:async () => {
      const { data } = await axios.get('https://hy-web.github.io/MyComponentPress/json/table-plus/table-plus1.json')
      return data
    },
    columns:[
      { title: '第一列', dataKey: 'a' },
      { title: '第二列', dataKey: 'b' },
      { title: '第三列', dataKey: 'c' }
    ]
  }
</script>

<template>
  <table-plus
    type="v1"
    :api="tableSet.api"
    :columns="tableSet.columns"
  />
</template>
